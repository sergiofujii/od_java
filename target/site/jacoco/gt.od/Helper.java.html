package gt.od;

import java.util.ArrayList;
import java.util.List;

/**
 *
 * @author calebemicael
 */
public class Helper {
    static List<String> tokensInLine(String line){
	String[] tokens = line.split("<|>|([a-z][A-Z][0-9])+");
	ArrayList<String> lista = new ArrayList<>();
	for(String token: tokens){
	    lista.add(token);
	}
	return lista;
    }
}
